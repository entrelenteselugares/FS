## Arquitetura da solução

```
Comprador paga
    ↓
Sistema verifica: e-mail do pedido == e-mail do usuário logado?
    ↓ sim
Escolhe: Público (90 dias) ou Privado (15 dias)
    ↓
Cron Job diário verifica expiração
    ↓
Arquivos excluídos permanentemente + notificação prévia
```

```typescript
import { Request, Response } from "express";
import { prisma } from "../lib/prisma";

// POST /api/orders/:id/access-type
// Comprador escolhe PUBLIC ou PRIVATE após pagamento
export async function chooseAccessType(req: Request, res: Response): Promise<void> {
  const { id } = req.params;
  const { accessType } = req.body;
  const userId = req.user!.userId;

  if (!["PUBLIC", "PRIVATE"].includes(accessType)) {
    res.status(400).json({ error: "Tipo inválido. Use PUBLIC ou PRIVATE." });
    return;
  }

  try {
    // Busca o pedido
    const order = await prisma.order.findFirst({
      where: { id },
      include: { event: true },
    });

    if (!order) {
      res.status(404).json({ error: "Pedido não encontrado." });
      return;
    }

    // ── Verificação LGPD ──────────────────────────────
    // O usuário logado deve ser o dono do pedido
    if (order.userId !== userId) {
      res.status(403).json({
        error: "Acesso negado. Apenas o titular do pedido pode definir a privacidade.",
      });
      return;
    }

    // Verifica se já foi escolhido (imutável após primeira escolha)
    if (order.accessType) {
      res.status(409).json({
        error: `Tipo de acesso já definido como ${order.accessType}. Esta escolha não pode ser alterada.`,
        accessType: order.accessType,
        accessExpiresAt: order.accessExpiresAt,
      });
      return;
    }

    // Verifica se o pagamento foi aprovado
    const aprovado = order.status === "APPROVED" || (order as any).status === "APROVADO";
    if (!aprovado) {
      res.status(403).json({ error: "Pagamento ainda não confirmado." });
      return;
    }

    // Calcula expiração baseado na escolha
    const now = new Date();
    const diasExpiracao = accessType === "PUBLIC" ? 90 : 15;
    const accessExpiresAt = new Date(now.getTime() + diasExpiracao * 24 * 60 * 60 * 1000);

    // Atualiza o pedido
    const updated = await prisma.order.update({
      where: { id },
      data: {
        accessType,
        accessChosenAt: now,
        accessExpiresAt,
      },
    });

    // Se PUBLIC, o evento fica visível no portfolio
    if (accessType === "PUBLIC") {
      await prisma.event.update({
        where: { id: order.eventId },
        data: { active: true },
      });
    }

    res.json({
      accessType: updated.accessType,
      accessExpiresAt: updated.accessExpiresAt,
      diasRestantes: diasExpiracao,
      mensagem: accessType === "PUBLIC"
        ? `Seu álbum ficará público por 90 dias. Faça o download antes de ${accessExpiresAt.toLocaleDateString("pt-BR")}.`
        : `Você tem 15 dias para fazer o download. Após ${accessExpiresAt.toLocaleDateString("pt-BR")} os arquivos serão excluídos permanentemente.`,
    });

  } catch (err) {
    console.error("chooseAccessType:", err);
    res.status(500).json({ error: "Erro ao definir tipo de acesso." });
  }
}

// GET /api/orders/:id/access-status
// Verifica status do acesso (usado pelo frontend para mostrar prazo)
export async function getAccessStatus(req: Request, res: Response): Promise<void> {
  const { id } = req.params;
  const userId = req.user!.userId;

  try {
    const order = await prisma.order.findFirst({
      where: { id, userId },
      include: {
        event: {
          select: {
            title: true, slug: true,
            lightroomUrl: true, driveUrl: true,
          },
        },
      },
    });

    if (!order) {
      res.status(404).json({ error: "Pedido não encontrado." });
      return;
    }

    const aprovado = order.status === "APPROVED" || (order as any).status === "APROVADO";

    // Verifica se expirou
    const now = new Date();
    const expirado = order.accessExpiresAt ? order.accessExpiresAt < now : false;
    const excluido = !!(order as any).deletedAt;

    if (excluido || expirado) {
      res.json({
        status: "EXPIRED",
        mensagem: "Este conteúdo foi excluído permanentemente conforme os termos de uso.",
        lightroomUrl: null,
        driveUrl: null,
      });
      return;
    }

    const diasRestantes = order.accessExpiresAt
      ? Math.ceil((order.accessExpiresAt.getTime() - now.getTime()) / (1000 * 60 * 60 * 24))
      : null;

    res.json({
      status: order.accessType ? "ACTIVE" : aprovado ? "PENDING_CHOICE" : "PENDING_PAYMENT",
      accessType: order.accessType,
      accessExpiresAt: order.accessExpiresAt,
      diasRestantes,
      // Links apenas se ativo e não expirado
      lightroomUrl: (order.accessType && !expirado) ? order.event.lightroomUrl : null,
      driveUrl: (order.accessType && !expirado) ? order.event.driveUrl : null,
      eventTitle: order.event.title,
      eventSlug: order.event.slug,
    });

  } catch (err) {
    console.error("getAccessStatus:", err);
    res.status(500).json({ error: "Erro ao verificar status." });
  }
}

// POST /api/admin/orders/:id/delete-media — exclusão manual pelo admin
export async function deleteMediaAdmin(req: Request, res: Response): Promise<void> {
  const { id } = req.params;

  try {
    await prisma.order.update({
      where: { id },
      data: { deletedAt: new Date() } as any,
    });
    res.json({ ok: true, mensagem: "Mídia marcada como excluída." });
  } catch {
    res.status(500).json({ error: "Erro ao excluir mídia." });
  }
}
```

### 3. Cron Job de expiração

**`src/jobs/expiration.job.ts`**

```typescript
import { prisma } from "../lib/prisma";

export async function runExpirationJob(): Promise<void> {
  const now = new Date();
  console.log(`[EXPIRATION JOB] Rodando em ${now.toISOString()}`);

  // ── 1. Envia aviso 3 dias antes da expiração ──────
  const tresDispara = new Date(now.getTime() + 3 * 24 * 60 * 60 * 1000);

  const proximosDeExpirar = await prisma.order.findMany({
    where: {
      accessType: { not: null },
      accessExpiresAt: {
        gte: now,
        lte: tresDispara,
      },
      warningsSent: { lt: 1 },
      deletedAt: null,
    } as any,
    include: {
      event: { select: { title: true } },
    },
  });

  for (const order of proximosDeExpirar) {
    const dias = Math.ceil(
      ((order.accessExpiresAt?.getTime() ?? 0) - now.getTime()) / (1000 * 60 * 60 * 24)
    );

    console.log(`[EXPIRATION JOB] Aviso: pedido ${order.id} expira em ${dias} dias`);

    // TODO: enviar e-mail/WhatsApp para order.buyerEmail
    // await sendExpirationWarning(order.buyerEmail, order.event.title, dias);

    await prisma.order.update({
      where: { id: order.id },
      data: { warningsSent: { increment: 1 } } as any,
    });
  }

  // ── 2. Marca como excluído os pedidos expirados ───
  const expirados = await prisma.order.findMany({
    where: {
      accessType: { not: null },
      accessExpiresAt: { lt: now },
      deletedAt: null,
    } as any,
    include: {
      event: { select: { title: true, id: true } },
    },
  });

  for (const order of expirados) {
    console.log(`[EXPIRATION JOB] Excluindo mídia do pedido ${order.id}`);

    await prisma.order.update({
      where: { id: order.id },
      data: { deletedAt: now } as any,
    });

    // Se era PUBLIC, verifica se ainda há outros pedidos ativos no evento
    if (order.accessType === "PUBLIC") {
      const outrosAtivos = await prisma.order.count({
        where: {
          eventId: order.event.id,
          accessType: "PUBLIC",
          deletedAt: null,
          accessExpiresAt: { gte: now },
        } as any,
      });

      // Se não há mais pedidos públicos ativos, desativa o evento do portfolio
      if (outrosAtivos === 0) {
        await prisma.event.update({
          where: { id: order.event.id },
          data: { active: false },
        });
      }
    }
  }

  console.log(`[EXPIRATION JOB] Concluído. ${proximosDeExpirar.length} avisos, ${expirados.length} exclusões.`);
}
```

## Resumo do fluxo completo

| Etapa | O que acontece |
|---|---|
| Pagamento aprovado | `hasPaid = true`, status `PENDING_CHOICE` |
| Modal aparece | Usuário escolhe Público ou Privado |
| Confirmação | `accessType` salvo, `accessExpiresAt` calculado |
| Público | Portfolio visível, 90 dias de prazo |
| Privado | Só o titular vê, 15 dias de prazo |
| 3 dias antes | Aviso de expiração enviado |
| Expiração | `deletedAt` marcado, links retornam `null` |

Após validar isso funcionando, partimos para o **sistema de curtidas e gamificação**.
